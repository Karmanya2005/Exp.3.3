02SOURCE_CODE
// Base class
class Person {
  constructor(name, age, email) {
    this.name  = name;
    this.age   = age;
    this.email = email;
  }

  getRole() { return "Person"; }

  getInfo() {
    return `[${this.getRole()}] ${this.name}, Age: ${this.age}`;
  }

  introduce() {
    return `Hi, I'm ${this.name} — a ${this.getRole()}.`;
  }
}
// Student extends Person
class Student extends Person {
  constructor(name, age, email, studentId, major, gpa) {
    super(name, age, email);
    this.studentId = studentId;
    this.major     = major;
    this.gpa       = gpa;
  }

  // @override
  getRole() { return "Student"; }

  // @override — calls super.getInfo()
  getInfo() {
    return `${super.getInfo()} | Major: ${this.major} | GPA: ${this.gpa}`;
  }

  study() {
    return `${this.name} is studying ${this.major}.`;
  }

  getGradeStatus() {
    if (this.gpa >= 3.7) return "Dean's List";
    if (this.gpa >= 3.0) return "Good Standing";
    return "Needs Improvement";
  }
}
// Teacher extends Person
class Teacher extends Person {
  constructor(name, age, email, empId, dept, subject, yrs) {
    super(name, age, email);
    this.employeeId = empId;
    this.department = dept;
    this.subject    = subject;
    this.yearsExp   = yrs;
  }

  // @override
  getRole() { return "Teacher"; }

  // @override — calls super.getInfo()
  getInfo() {
    return `${super.getInfo()} | ${this.department} | ${this.yearsExp}yr exp`;
  }

  teach() {
    return `${this.name} is teaching ${this.subject}.`;
  }

  getSeniorityLevel() {
    if (this.yearsExp >= 15) return "Senior Faculty";
    if (this.yearsExp >= 5)  return "Associate";
    return "Junior Faculty";
  }
}
// Polymorphism — same call, different behavior per class
const people = [
  new Student("Alice", 20, "alice@uni.edu", "S-001", "CS", 3.9),
  new Teacher("Dr. Chen", 45, "chen@uni.edu", "T-001", "Eng", "Algorithms", 18),
  new Student("Marcus", 22, "marcus@uni.edu", "S-002", "Math", 3.2),
];

people.forEach(p => {
  console.log(p.getInfo());    // polymorphic — each class responds differently
  console.log(p.introduce());  // inherited, uses overridden getRole()
  console.log(p instanceof Person); // → true for all
});
